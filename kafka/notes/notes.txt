I-- Mysql config (my-custom.cnf -> /etc/mysql/conf.d/my-custom.cnf)
# under [mysqld]

# 0. disable
default_authentication_plugin=mysql_native_password

# 1. Enable Binary Logging
log-bin=mysql-bin
binlog_format=ROW
binlog_row_image=FULL
binlog_rows_query_log_events=ON

# 2. Set Retention (7 days is standard to avoid disk fill-up)
binlog_expire_logs_seconds=604800

# 3. Server ID (Must be unique within your replication topology)
server-id=1

# 4. GTID Mode (Recommended for recovery and consistency)
gtid_mode=ON
enforce_gtid_consistency=ON


II-- user config (nanp)
GRANT SELECT, RELOAD, SHOW DATABASES, REPLICATION SLAVE, REPLICATION CLIENT ON *.* TO 'nanp'@'%';
FLUSH PRIVILEGES;

-- permisisons d'un utilisateur
SHOW GRANTS;
SHOW GRANTS FOR CURRENT_USER;
SHOW GRANTS FOR 'nom_utilisateur'@'localhost';

-- List all users
SELECT User, Host FROM mysql.user;

III-- check variables values (you can replace 'log_bin' with 'binlog_format' and so on)
SELECT variable_value as "BINARY LOGGING STATUS (log-bin) ::"
FROM performance_schema.global_variables WHERE variable_name='log_bin';

IV-- for server id and uuid (go to my-custom.cnf to change it)
SELECT @@server_id;
SELECT @@server_uuid;

V-- check GTID (go to my-custom.cnf to change it)
show global variables like '%GTID%';
